## Create the store

```ts
import { Pool } from "pg";
import { createRawSqlVectorStore } from "@unrag/store/raw-sql";

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
export const store = createRawSqlVectorStore(pool);
```
